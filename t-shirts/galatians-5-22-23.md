def is_fruit(piece):
    match piece:
        case 💗:  # Love
        case 🤩:  # Joy
        case 🕊️:  # Peace
        case 🙏:  # Patience
        case 😊:  # Kindness
        case 🎁:  # Goodness
        case 🤝:  # Faithfulness
        case 🧘:  # Gentleness
        case ⛓️:  # Self-control
            return True
        case _:
            return False
# Galatians 5:22-23
